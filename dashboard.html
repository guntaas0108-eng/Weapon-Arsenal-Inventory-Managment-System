schema.sql
CREATE TABLE Category (
    category_id    NUMBER(4)    PRIMARY KEY,
    category_name  VARCHAR2(40) NOT NULL,
    description    VARCHAR2(200),
    class_level    VARCHAR2(20)
);

CREATE TABLE Weapon (
    weapon_id    NUMBER(6)    PRIMARY KEY,
    name         VARCHAR2(50) NOT NULL,
    category_id  NUMBER(4)    REFERENCES Category(category_id),
    serial_no    VARCHAR2(30) UNIQUE NOT NULL,
    quantity     NUMBER(5)    DEFAULT 1 CHECK(quantity >= 0),
    condition    VARCHAR2(20) CHECK(condition IN
                 ('Operational', 'Maintenance', 'Decommissioned')),
    location     VARCHAR2(40)
);

CREATE TABLE Personnel (
    personnel_id  NUMBER(6)    PRIMARY KEY,
    name          VARCHAR2(80) NOT NULL,
    rank          VARCHAR2(30),
    unit          VARCHAR2(50),
    clearance     VARCHAR2(20),
    contact       VARCHAR2(15)
);

CREATE TABLE Issue (
    issue_id      NUMBER(8)   PRIMARY KEY,
    weapon_id     NUMBER(6)   REFERENCES Weapon(weapon_id),
    personnel_id  NUMBER(6)   REFERENCES Personnel(personnel_id),
    issue_date    DATE        DEFAULT SYSDATE,
    return_date   DATE,
    status        VARCHAR2(12) DEFAULT 'ISSUED'
);

CREATE TABLE Maintenance (
    maintenance_id  NUMBER(8)   PRIMARY KEY,
    weapon_id       NUMBER(6)   REFERENCES Weapon(weapon_id),
    tech_id         VARCHAR2(20),
    date_logged     DATE        DEFAULT SYSDATE,
    type            VARCHAR2(40),
    outcome         VARCHAR2(20),
    next_due        DATE
);
issue_weapon.sql — Stored Procedure
CREATE OR REPLACE PROCEDURE issue_weapon (
    p_weapon_id     IN Weapon.weapon_id%TYPE,
    p_personnel_id  IN Personnel.personnel_id%TYPE
) AS
    v_qty   Weapon.quantity%TYPE;
    v_cond  Weapon.condition%TYPE;
BEGIN
    SELECT quantity, condition
    INTO   v_qty, v_cond
    FROM   Weapon
    WHERE  weapon_id = p_weapon_id FOR UPDATE;

    IF v_qty < 1 OR v_cond != 'Operational' THEN
        RAISE_APPLICATION_ERROR(-20001, 'Weapon unavailable');
    END IF;

    INSERT INTO Issue (issue_id, weapon_id, personnel_id, issue_date, status)
    VALUES (issue_seq.NEXTVAL, p_weapon_id, p_personnel_id, SYSDATE, 'ISSUED');

    UPDATE Weapon SET quantity = quantity - 1
    WHERE  weapon_id = p_weapon_id;

    COMMIT;
EXCEPTION
    WHEN OTHERS THEN ROLLBACK; RAISE;
END;
/
-- Usage: EXEC issue_weapon(3, 2);
triggers.sql
-- TRIGGER 1: Schedule maintenance after 180 days
CREATE OR REPLACE TRIGGER trg_maintenance_check
AFTER INSERT ON Issue FOR EACH ROW
DECLARE v_last DATE;
BEGIN
    SELECT MAX(date_logged) INTO v_last
    FROM Maintenance WHERE weapon_id = :NEW.weapon_id;

    IF v_last IS NULL OR (SYSDATE - v_last) > 180 THEN
        INSERT INTO Maintenance
            (maintenance_id, weapon_id, type, outcome, next_due)
        VALUES (maint_seq.NEXTVAL, :NEW.weapon_id,
                'SCHEDULED_INSPECTION', 'PENDING', SYSDATE+14);
    END IF;
END; /

-- TRIGGER 2: Flag overdue returns automatically
CREATE OR REPLACE TRIGGER trg_overdue_flag
BEFORE UPDATE ON Issue FOR EACH ROW
BEGIN
    IF :NEW.return_date IS NULL
       AND (SYSDATE - :OLD.issue_date) > 30
       AND :NEW.status != 'RETURNED'
    THEN :NEW.status := 'OVERDUE';
    END IF;
END; /
cursor.sql — Inventory Report
DECLARE
    CURSOR c_inv IS
        SELECT w.weapon_id, w.name, c.category_name,
               w.quantity, w.condition, w.location
        FROM   Weapon w
        JOIN   Category c ON w.category_id = c.category_id
        ORDER BY c.category_name, w.name;
    v_rec   c_inv%ROWTYPE;
    v_total NUMBER := 0;
BEGIN
    DBMS_OUTPUT.PUT_LINE('=== WAIMS INVENTORY REPORT ===');
    OPEN c_inv;
    LOOP
        FETCH c_inv INTO v_rec;
        EXIT WHEN c_inv%NOTFOUND;
        v_total := v_total + v_rec.quantity;
        DBMS_OUTPUT.PUT_LINE(
            RPAD(v_rec.name, 20) || ' | ' ||
            RPAD(v_rec.category_name, 15) ||
            ' | Qty:' || v_rec.quantity ||
            ' | ' || v_rec.condition);
    END LOOP;
    DBMS_OUTPUT.PUT_LINE('Total types : ' || c_inv%ROWCOUNT);
    DBMS_OUTPUT.PUT_LINE('Total units : ' || v_total);
    CLOSE c_inv;
END; /
functions.sql
-- FUNCTION 1: Operational stock for a category
CREATE OR REPLACE FUNCTION get_category_stock(
    p_cat_id IN NUMBER
) RETURN NUMBER IS
    v_total NUMBER;
BEGIN
    SELECT NVL(SUM(quantity), 0) INTO v_total
    FROM   Weapon
    WHERE  category_id = p_cat_id AND condition = 'Operational';
    RETURN v_total;
END; /

-- Usage:
SELECT category_name,
       get_category_stock(category_id) AS op_units
FROM Category;

-- FUNCTION 2: Is a weapon overdue for maintenance?
CREATE OR REPLACE FUNCTION is_maint_due(
    p_weapon_id IN NUMBER
) RETURN VARCHAR2 IS
    v_last DATE;
BEGIN
    SELECT MAX(date_logged) INTO v_last
    FROM   Maintenance
    WHERE  weapon_id = p_weapon_id;

    IF v_last IS NULL OR (SYSDATE - v_last) > 180
    THEN RETURN 'YES — OVERDUE';
    ELSE RETURN 'NO — OK';
    END IF;
END; /

-- Usage:
SELECT name, is_maint_due(weapon_id) AS maint_due
FROM   Weapon;

joins_subqueries.sql — JOINs & Subqueries
-- ══ INNER JOIN ══════════════════════════════════════
-- All issued weapons with officer names
SELECT w.name AS weapon, p.name AS officer,
       p.rank, i.issue_date, i.status
FROM   Issue i
JOIN   Weapon w    ON i.weapon_id    = w.weapon_id
JOIN   Personnel p ON i.personnel_id = p.personnel_id
ORDER BY i.issue_date DESC;

-- ══ LEFT JOIN ════════════════════════════════════════
-- All weapons, including those never issued
SELECT w.name, w.condition,
       NVL(TO_CHAR(i.issue_date, 'DD-MON-YYYY'), 'Never Issued') AS last_issued
FROM   Weapon w
LEFT JOIN Issue i ON w.weapon_id = i.weapon_id;

-- ══ JOIN with Category ═══════════════════════════════
-- Full inventory report with category name
SELECT w.weapon_id, w.name, c.category_name,
       w.serial_no, w.quantity, w.condition, w.location
FROM   Weapon w
JOIN   Category c ON w.category_id = c.category_id
ORDER BY c.category_name, w.name;

-- ══ SUBQUERY ═════════════════════════════════════════
-- Weapons that have never been maintained
SELECT name, condition
FROM   Weapon
WHERE  weapon_id NOT IN (
    SELECT DISTINCT weapon_id FROM Maintenance
);

-- Weapons issued by officers with TOP SECRET clearance
SELECT w.name, w.quantity
FROM   Weapon w
WHERE  w.weapon_id IN (
    SELECT i.weapon_id
    FROM   Issue i
    JOIN   Personnel p ON i.personnel_id = p.personnel_id
    WHERE  p.clearance = 'TOP SECRET'
);

-- Correlated subquery: weapons below average quantity
SELECT name, quantity
FROM   Weapon
WHERE  quantity < (
    SELECT AVG(quantity) FROM Weapon
    WHERE  condition = 'Operational'
);
aggregates_views.sql — Aggregate Functions & Views
-- ══ AGGREGATE FUNCTIONS ══════════════════════════════
-- Total weapon units in arsenal
SELECT COUNT(*) AS total_types,
       SUM(quantity) AS total_units,
       AVG(quantity) AS avg_stock,
       MAX(quantity) AS max_stock,
       MIN(quantity) AS min_stock
FROM Weapon
WHERE condition = 'Operational';

-- ══ GROUP BY ═════════════════════════════════════════
-- Units per category
SELECT c.category_name,
       COUNT(w.weapon_id) AS weapon_types,
       SUM(w.quantity)    AS total_units
FROM   Weapon w
JOIN   Category c ON w.category_id = c.category_id
GROUP BY c.category_name
ORDER BY total_units DESC;

-- Issues per officer
SELECT p.name, p.rank,
       COUNT(i.issue_id) AS total_Issues
FROM   Issue i
JOIN   Personnel p ON i.personnel_id = p.personnel_id
GROUP BY p.name, p.rank
ORDER BY total_Issues DESC;

-- ══ HAVING ═══════════════════════════════════════════
-- Categories with more than 30 total units
SELECT c.category_name, SUM(w.quantity) AS total_units
FROM   Weapon w
JOIN   Category c ON w.category_id = c.category_id
GROUP BY c.category_name
HAVING SUM(w.quantity) > 30
ORDER BY total_units DESC;

-- Officers with 2 or more active Issues
SELECT p.name, COUNT(i.issue_id) AS active_Issues
FROM   Issue i
JOIN   Personnel p ON i.personnel_id = p.personnel_id
WHERE  i.status IN ('ISSUED', 'OVERDUE')
GROUP BY p.name
HAVING COUNT(i.issue_id) >= 2;

-- ══ VIEWS ════════════════════════════════════════════
-- View: Active issue summary
CREATE OR REPLACE VIEW vw_active_Issues AS
SELECT i.issue_id, w.name AS weapon,
       p.name AS officer, p.rank,
       i.issue_date, i.status
FROM   Issue i
JOIN   Weapon w    ON i.weapon_id    = w.weapon_id
JOIN   Personnel p ON i.personnel_id = p.personnel_id
WHERE  i.status IN ('ISSUED', 'OVERDUE');

-- View: Operational inventory
CREATE OR REPLACE VIEW vw_operational_stock AS
SELECT w.weapon_id, w.name, c.category_name,
       w.quantity, w.location
FROM   Weapon w
JOIN   Category c ON w.category_id = c.category_id
WHERE  w.condition = 'Operational';

-- View: Maintenance due report
CREATE OR REPLACE VIEW vw_maintenance_due AS
SELECT w.name, m.tech_id, m.type,
       m.next_due,
       (SYSDATE - m.next_due) AS days_overdue
FROM   Maintenance m
JOIN   Weapon w ON m.weapon_id = w.weapon_id
WHERE  m.next_due < SYSDATE
  AND  m.outcome != 'COMPLETED';

-- Using a view
SELECT * FROM vw_active_Issues;
SELECT * FROM vw_operational_stock WHERE quantity < 10;
transactions.sql — COMMIT / ROLLBACK / SAVEPOINT
-- ══ TRANSACTION: Issue weapon with rollback on failure ══
BEGIN
    SAVEPOINT before_issue;

    -- Step 1: Validate weapon availability
    DECLARE
        v_qty    NUMBER;
        v_cond   VARCHAR2(20);
    BEGIN
        SELECT quantity, condition
        INTO   v_qty, v_cond
        FROM   Weapon
        WHERE  weapon_id = 3 FOR UPDATE;

        IF v_qty < 1 OR v_cond != 'Operational' THEN
            ROLLBACK TO before_issue;
            RAISE_APPLICATION_ERROR(-20001, 'Weapon not available');
        END IF;

        -- Step 2: Insert issue record
        INSERT INTO Issue
            (issue_id, weapon_id, personnel_id, issue_date, status)
        VALUES
            (issue_seq.NEXTVAL, 3, 2, SYSDATE, 'ISSUED');

        SAVEPOINT after_insert;

        -- Step 3: Decrement stock
        UPDATE Weapon
        SET    quantity = quantity - 1
        WHERE  weapon_id = 3;

        -- All steps succeeded — commit
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Transaction committed successfully.');

    EXCEPTION
        WHEN OTHERS THEN
            ROLLBACK TO before_issue;
            DBMS_OUTPUT.PUT_LINE('Transaction rolled back: ' || SQLERRM);
    END;
END;
/

-- ══ TRANSACTION: Bulk restock with SAVEPOINT ══════════
BEGIN
    SAVEPOINT restock_start;

    UPDATE Weapon SET quantity = quantity + 20 WHERE weapon_id = 1;
    SAVEPOINT after_weapon1;

    UPDATE Weapon SET quantity = quantity + 15 WHERE weapon_id = 2;
    SAVEPOINT after_weapon2;

    UPDATE Weapon SET quantity = quantity + 10 WHERE weapon_id = 3;

    COMMIT; -- All updates committed together

EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK TO restock_start; -- Roll back all if any fails
        RAISE;
END;
/

-- ══ ACID PROPERTIES DEMONSTRATED ════════════════════
-- Atomicity  : All steps commit together or all rollback
-- Consistency: CHECK constraints prevent invalid data
-- Isolation  : FOR UPDATE locks rows during transaction
-- Durability : COMMIT writes permanently to storage
Normalization
// 1NF → 2NF → 3NF with functional dependencies
// FIRST NORMAL FORM (1NF) — All attributes atomic, no repeating groups
A relation is in 1NF if every attribute contains only atomic (indivisible) values and each row is unique.
Before 1NF — violation example
WeaponID | WeaponName | Categories (VIOLATION — multi-valued)
---------|------------|----------------------------------
  1      | AK-47      | Assault Rifle, SMG        ← NOT atomic
  2      | Glock 19   | Pistol, Sidearm           ← NOT atomic
After 1NF — WAIMS Weapon table
WeaponID | WeaponName | CategoryID | SerialNo       | Quantity | Condition
---------|------------|------------|----------------|----------|-----------
  1      | AK-47      |     1      | AK-2021-001    |    45    | Operational
  2      | Glock 19   |     2      | GL-2022-007    |    80    | Operational
-- ✓ All values atomic, each row uniquely identified by WeaponID (PK)
// SECOND NORMAL FORM (2NF) — No partial dependencies on composite keys
A relation is in 2NF if it is in 1NF and every non-key attribute is fully functionally dependent on the entire primary key.
Before 2NF — partial dependency violation
Issue (IssueID, WeaponID, PersonnelID, WeaponName, PersonnelRank, IssueDate)
-- Composite PK = (WeaponID, PersonnelID)
-- WeaponName depends only on WeaponID     ← PARTIAL DEPENDENCY
-- PersonnelRank depends only on PersonnelID ← PARTIAL DEPENDENCY
After 2NF — decomposed into separate tables
Weapon    (WeaponID PK, WeaponName, CategoryID, Quantity, Condition, Location)
Personnel (PersonnelID PK, Name, Rank, Unit, Clearance, Contact)
Issue     (IssueID PK, WeaponID FK, PersonnelID FK, IssueDate, ReturnDate, Status)
-- ✓ Each non-key attribute depends on the FULL primary key of its table
// THIRD NORMAL FORM (3NF) — No transitive dependencies
A relation is in 3NF if it is in 2NF and no non-key attribute depends on another non-key attribute (no transitive dependency).
Before 3NF — transitive dependency
Weapon (WeaponID, WeaponName, CategoryID, CategoryName, CategoryClass)
-- WeaponID → CategoryID → CategoryName  ← TRANSITIVE DEPENDENCY
-- CategoryName depends on CategoryID, not directly on WeaponID
After 3NF — Category extracted as separate table
Category (CategoryID PK, CategoryName, Description, ClassLevel)
Weapon   (WeaponID PK, WeaponName, CategoryID FK, Quantity, Condition, Location)
-- ✓ No non-key attribute depends on another non-key attribute
-- ✓ All 5 tables satisfy 3NF — minimal redundancy achieved
// FUNCTIONAL DEPENDENCIES — FINAL SCHEMA
CategoryID → CategoryName, Description, ClassLevel
WeaponID → WeaponName, CategoryID, SerialNo, Quantity, Condition, Location
PersonnelID → Name, Rank, Unit, Clearance, Contact
IssueID → WeaponID, PersonnelID, IssueDate, ReturnDate, Status
MaintenanceID → WeaponID, TechID, DateLogged, Type, Outcome, NextDue
